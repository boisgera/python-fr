Solution

Il semble naturel de repr√©senter les sommets comme un ensemble de paires d‚Äôentiers, les ar√™tes comme un ensemble de paires de sommets et les poids comme un dictionnaire ayant comme cl√©s les sommets et comme valeur unique l‚Äôentier 1.

def maze_to_graph(maze):
    vertices = set(maze)
    edges = set()
    weights = {}
    for vertex in vertices:
        x, y = vertex
        for (dx, dy) in [(-1, 0), (0, -1), (1, 0), (0, 1)]:
            neighbor = (x + dx, y + dy)
            if neighbor in vertices:
                edge = (vertex, neighbor)
                edges.add(edge)
                weights[edge] = 1
    return (vertices, edges, weights)

def shortest_path_from(maze, source): 
    vertices, edges, weight = maze_to_graph(maze)
    distance, path = {}, {}
    todo = {source}
    distance[source] = 0
    path[source] = [source]
    while todo:
        current = todo.pop()
        neighbors = {
            v for v in vertices 
            if (current, v) in edges
        }
        for n in neighbors:
            d = distance[current] + weight[(current, n)]
            if d < distance.get(n, math.inf):

                path[n] = path[current] + [n]
                todo.add(n)
    return path
